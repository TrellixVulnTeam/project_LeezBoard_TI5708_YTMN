63 static inline void lib_imt_copy(const void *src, void *dst, size_t size)
64 {
65  /* use volatile to prevent compiler from optimizing away subsequent reads */
66  volatile uint32_t pend;
67 
68  /* wait untill there is no pending transfer */
69  while( (pend = CSL_FEXT(hIdma->IDMA1_STAT, CGEM_IDMA1_STAT_PEND) ) );
70 
71  /* hard code the following configuration:
72  - Block fill: 0
73  - DSP interrupt enable: 0
74  - Transfer priority: 7 (lowest)
75  */
76  hIdma->IDMA1_SOURCE = (uint32_t)src;
77  hIdma->IDMA1_DEST = (uint32_t)dst;
78  hIdma->IDMA1_COUNT = CSL_FMK(CGEM_IDMA1_COUNT_PRI, (uint32_t)7) |
79  CSL_FMK(CGEM_IDMA1_COUNT_INT, (uint32_t)0) |
80  CSL_FMK(CGEM_IDMA1_COUNT_FILL, (uint32_t)0) |
81  CSL_FMK(CGEM_IDMA1_COUNT_COUNT, size);
82  return;
83 }

99 static inline void lib_imt_set(uint32_t pattern, void *dst, size_t size)
100 {
101  /* use volatile to prevent compiler from optimizing away subsequent reads */
102  volatile uint32_t pend;
103 
104  /* wait untill there is no pending transfer */
105  while( (pend = CSL_FEXT(hIdma->IDMA1_STAT, CGEM_IDMA1_STAT_PEND) ) );
106 
107  /* hard code the following configuration:
108  - Block fill: 1
109  - DSP interrupt enable: 0
110  - Transfer priority: 7 (lowest)
111  */
112  hIdma->IDMA1_SOURCE = pattern;
113  hIdma->IDMA1_DEST = (uint32_t)dst;
114  hIdma->IDMA1_COUNT = CSL_FMK(CGEM_IDMA1_COUNT_PRI, (uint32_t)7) |
115  CSL_FMK(CGEM_IDMA1_COUNT_INT, (uint32_t)0) |
116  CSL_FMK(CGEM_IDMA1_COUNT_FILL, (uint32_t)1) |
117  CSL_FMK(CGEM_IDMA1_COUNT_COUNT, size);
118  return;
119 }
